```ts
export type DlcColor = "green" | "orange" | "red" | "grey";

/** ISO date string (YYYY-MM-DD), production + 5 days. */
export function computeBrocheDlc(dateProduction: string): string {
  // dateProduction may arrive as 'YYYY-MM-DD' or full ISO; we operate UTC-only.
  const base = new Date(`${dateProduction.slice(0, 10)}T00:00:00.000Z`);
  if (Number.isNaN(base.getTime())) throw new RangeError(`Invalid dateProduction: ${dateProduction}`);
  base.setUTCDate(base.getUTCDate() + 5);
  return base.toISOString().slice(0, 10);
}

/**
 * PRD ¬ß6 thresholds (date-only, UTC):
 * - >5 days remaining -> "green"
 * - 2-5 days -> "orange"
 * - 0-1 days -> "red"
 * - expired (negative) -> "grey"
 */
export function dlcColor(dlc: string, today: Date): DlcColor {
  const dlcDay = new Date(`${dlc.slice(0, 10)}T00:00:00.000Z`);
  const todayDay = new Date(`${today.toISOString().slice(0, 10)}T00:00:00.000Z`);
  if (Number.isNaN(dlcDay.getTime())) throw new RangeError(`Invalid dlc: ${dlc}`);
  const days = Math.floor((dlcDay.getTime() - todayDay.getTime()) / 86_400_000);
  if (days < 0) return "grey";
  if (days < 2) return "red";
  if (days <= 5) return "orange";
  return "green";
}
```

Critical thresholds ‚ÄĒ DO NOT round; strip time-of-day from both sides via UTC-midnight reconstruction so a 4 PM `today` comparing against a same-day DLC returns 0 days (red), not -0.7 days (grey). ~50 lines.

```ts
export const useTraceabilityStore = create<TraceabilityStore>()(
  persist(
    (set, get) => ({
      ...initialState,

      addRawMaterial: (rm) => set((s) => ({ rawMaterials: [...s.rawMaterials, rm] })),
      updateRawMaterial: (id, patch) =>
        set((s) => ({
          rawMaterials: s.rawMaterials.map((x) => (x.id === id ? { ...x, ...patch } : x)),
        })),
      deleteRawMaterial: (id) =>
        set((s) => ({ rawMaterials: s.rawMaterials.filter((x) => x.id !== id) })),

      // ... repeat for recipes / productionOrders / finishedProducts / customers / deliveries
      //     using the same three-line pattern. Total: 18 actions.

      seedIfEmpty: () => {
        const s = get();
        if (s.rawMaterials.length === 0 && s.recipes.length === 0 && s.customers.length === 0) {
          const seed = buildSeed();
          set({
            rawMaterials: seed.rawMaterials,
            recipes: seed.recipes,
            productionOrders: seed.productionOrders,
            finishedProducts: seed.finishedProducts,
            customers: seed.customers,
            deliveries: seed.deliveries,
          });
        }
      },

      resetToSeed: () => {
        // Clear the persisted snapshot so a hard reload would also seed cleanly.
        if (typeof window !== "undefined") {
          window.localStorage.removeItem(STORAGE_KEY);
        }
        const seed = buildSeed();
        set({
          rawMaterials: seed.rawMaterials,
          recipes: seed.recipes,
          productionOrders: seed.productionOrders,
          finishedProducts: seed.finishedProducts,
          customers: seed.customers,
          deliveries: seed.deliveries,
        });
      },

      setHasHydrated: (v) => set({ hasHydrated: v }),
    }),
    {
      name: STORAGE_KEY,
      storage: createJSONStorage(() => localStorage),
      // Persist data only ‚ÄĒ never persist the hydration flag.
      partialize: (state) => ({
        rawMaterials: state.rawMaterials,
        recipes: state.recipes,
        productionOrders: state.productionOrders,
        finishedProducts: state.finishedProducts,
        customers: state.customers,
        deliveries: state.deliveries,
      }),
      onRehydrateStorage: () => (state) => {
        // Fires after persisted state is loaded back into the store on the client.
        state?.setHasHydrated(true);
      },
    },
  ),
);
```
